Функция decode : const decode = (inp)=>
    inp
    .replace(
      /.\d+/g,
      (x) => new Array(+x.slice(1)).fill(x.includes(".") ? 1 : 0).join("") + "-"
    )
    .split("-")
    .filter(x=>x !== '')
     .map(x=>[...x])
     .map(x=>x.length===1 ? '.' : `${(x.length -2)/ 2}`)
     .join('')
     .split('.')
     .map(x=>+x)
     .flatMap((value,index,array)=> index % 2 === 0 ? [[value,array[index+1]]] : [])
     .sort((a,b)=>a[1] - b[1])
     .map((x,i)=>String.fromCharCode(x[0]))
